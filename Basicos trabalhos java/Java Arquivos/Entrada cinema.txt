import javax.swing.JOptionPane;
public class EntradadeCinema {
public static void main(String[]args){
	int idade,diadasemana;
	idade=Integer.parseInt(JOptionPane.showInputDialog(null,"Entre idade "));
	diadasemana=Integer.parseInt(JOptionPane.showInputDialog(null,"Entre dia - (1=dom,2=seg,3=terça,4=quar,5=quin,6=sex,7=sabado)"));
	float preço=calculaPreço(idade,diadasemana);
	JOptionPane.showInputDialog(null,"O preço da entrada é  "+preço+"reais","Entrada de Cinema",1);
	}
static float calculaPreço(int idade,int diadasemana){
	float preço;
	if ((diadasemana>=3)&& (diadasemana<5))
		preço=10;
		else{
			if ((idade<=14)||(idade>=65))
				preço=10;
			else
				preço=20;
		}
		return preço;
}
}
